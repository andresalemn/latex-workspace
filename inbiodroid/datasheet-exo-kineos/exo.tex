\section{Descripción General}
\begin{itemize}
\item El Exo-Kineos es un sistema biomecatrónico (exoesqueleto) de evaluación y análisis muscular diseñado para medir el esfuerzo muscular (actividad EMG) y la cinemática del movimiento (posición y velocidad articular) en tiempo real.
\item Exo-Kineos EK: Evaluación muscular para la articulación del codo (brazo-antebrazo) o la rodilla (pierna).
\item Exo-Kineos HS: Análisis muscular para la articulación compleja del hombro o la cadera.
\end{itemize}

\section{Funcionamiento}
\begin{itemize}
 \item Principios de Funcionamiento:
 \begin{itemize}
    \item Electromiografía (EMG): Mide el potencial eléctrico de los músculos (actividad muscular) mediante sensores de superficie MyoWare 2.0.
    \item Cinemática: Cuantifica el posición y la velocidad del movimiento articular mediante encoders de alta precisión integrados en los eslabones del exoesqueleto.

 \end{itemize}
\end{itemize}

\section{Aplicacion de Escritorio}
\begin{itemize}
\item Kineos-View
    \begin{itemize}
        \item Tipo: Aplicación de escritorio para Windows (archivo .exe).
        \item Funcionalidad: Visualización en tiempo real de gráficas de Posición Angular, Velocidad Angular y señales EMG en bruto. Almacenamiento de datos.
        \item Gráficas por Modelo:
        \begin{itemize}
            \item Exo-Kineos EK (Codo/Rodilla): 4 gráficas (1 Posición, 1 Velocidad, 2 EMG).
            \item Exo-Kineos HS (Hombro/Cadera): 9 gráficas (3 Posición, 3 Velocidad, 3 EMG).
        \end{itemize}

    \end{itemize}
\end{itemize}
